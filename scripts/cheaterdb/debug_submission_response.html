Problem: 2141C - Minimum on Subarrays

#include <iostream> #include <vector> #include <string> int main() { // Fast I/O std::ios_base::sync_with_stdio(false); std::cin.tie(NULL); int n; std::cin >> n; // A vector to store the sequence of commands. std::vector<std::string> commands; // Iterate through each possible starting element of a subarray, from a[n-1] down to a[0]. for (int i = n - 1; i >= 0; --i) { // At the beginning of this loop, the data structure is assumed to contain [a[i+1], ..., a[n-1]]. // For i = n-1, it's empty. // Add a[i] to the front. The structure now contains [a[i], ..., a[n-1]]. commands.push_back("pushfront a[" + std::to_string(i) + "]"); // The current state of the structure represents the subarray a[i...n-1]. // We find its minimum. commands.push_back("min"); // Now, we generate all shorter subarrays starting at `i` by popping from the back. for (int j = n - 1; j > i; --j) { commands.push_back("popback"); // After popback, the structure represents subarray a[i...j-1]. commands.push_back("min"); } // At this point, the structure only contains [a[i]]. // To prepare for the next iteration (i-1), we need to restore the elements // a[i+1]...a[n-1] so that the structure contains [a[i], a[i+1], ..., a[n-1]]. // This step is not needed for the last iteration (i=0). if (i > 0) { for (int j = i + 1; j < n; ++j) { commands.push_back("pushback a[" + std::to_string(j) + "]"); } } } // Output the total number of commands and the commands themselves. std::cout << commands.size() << std::endl; for (const auto& cmd : commands) { std::cout << cmd << std::endl; } return 0; }
